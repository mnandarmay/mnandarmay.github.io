We then define two functions, one to calculate average, and the other to calculate sensitivity, specificity, precision, recall, and f1.

In[4]:
def avg(l):
    return sum(l) / float(len(l))

def calculate_metrics(y_true, y_pred):
    tn, fp, fn, tp = confusion_matrix(y_true, y_pred).ravel()
    sensitivity = tp / (tp + fn) if (tp + fn) > 0 else 0
    specificity = tn / (tn + fp) if (tn + fp) > 0 else 0
    precision = precision_score(y_true, y_pred)
    recall = recall_score(y_true, y_pred)
    f1 = f1_score(y_true, y_pred)
    return sensitivity, specificity, precision, recall, f1

Next we define fitness function. Fitness function is the evaluation metrics for each subset ('chromosome') and it is based on cross-validated performane of a classifier using the selected features.

The input value for the function is individual (list of binary values in a certain subset), X is feature matrix, y is the target vector, and clf is the classifier to be evaluated.

The logic behind the code:

  • first we check if individual contains at least one feature, if it is all zero, then fitness function should be zero.
  • 'cols' is list of column indices corresponding to features that are not selected ('0').
  • then we create subset of feature matrix X ('X_subset') by deleting columns that are not selected (so only columns that are selected are left in X).
  • we then evaluate the classfier using 5-fold cross validation on 'X_subset'. It will return array of scores for each fold but we compute the average score.
  • the whole getFitness function return the average cv score as a tuple.
In[5]:
def getFitness(individual, X, y, clf):
    if individual.count(0) != len(individual):
        cols = [index for index in range(len(individual)) if individual[index] == 0]
        X_subset = np.delete(X, cols, axis=1)
        return (avg(cross_val_score(clf, X_subset, y, cv=5)),)
    else:
        return (0,)

The next chunk of code sets up and execute the genetic algorithm (GA) using Distributed Evolutionary Algorithms in Python.

Function parameters:

  • X: The feature matrix.
  • y: The target vector.
  • clf: The classifier to be optimized.
  • n_population: Number of individuals in the population.
  • n_generation: Number of generations for GA runs.
  • cxpb: Crossover probability.
  • mutpb: Mutation probability.
  • elite_count: Number of elite individuals retained for each generation.
  • stall_gen_limit: Maximum number of generations with no improvement before stopping.

We start by writing code to avoid overwriting existing class because the first iteration of the code resulting in too many warning messages.

In addition to that, to prevent wasting computations on stagnant populations, we implement stall generation limit to stop the algorithm early if no improvemkent is observed over several generations.

In[6]:
def geneticAlgorithm(X, y, clf, n_population, n_generation, cxpb, mutpb, elite_count, stall_gen_limit, seed=None):
    if seed is not None:
        random.seed(seed)
        np.random.seed(seed)
        deap_random = random.Random(seed)  # Create a separate instance of the random generator
    else:
        deap_random = random
        
    # Avoid overwriting existing classes
    if not hasattr(creator, "FitnessMax"):
        creator.create("FitnessMax", base.Fitness, weights=(1.0,))
    if not hasattr(creator, "Individual"):
        creator.create("Individual", list, fitness=creator.FitnessMax)

    toolbox = base.Toolbox()  # Initializes toolbox for the GA
    toolbox.register("attr_bool", deap_random.randint, 0, 1)  # Register function to create random binary attribute
    toolbox.register("individual", tools.initRepeat, creator.Individual, toolbox.attr_bool, X.shape[1])  # Function to create an individual with list of binary attributes
    toolbox.register("population", tools.initRepeat, list, toolbox.individual)  # Register a function to create population of individuals
    toolbox.register("evaluate", getFitness, X=X, y=y, clf=clf)  # Register the fitness evaluation function
    toolbox.register("mate", tools.cxUniform, indpb=0.5)  # Register the crossover operation with consistent rate
    toolbox.register("mutate", tools.mutFlipBit, indpb=0.033)  # Register the mutation operation with consistent rate
    toolbox.register("select", tools.selTournament, tournsize=3)  # Registers the selection operation (tournament selection) type
    toolbox.register("elitism", tools.selBest, k=elite_count)  # Register the elitism operation

    pop = toolbox.population(n=n_population)  # Creates initial population
    hof = tools.HallOfFame(n_population * n_generation)  # Initializes hall of fame to keep track of best subset of features
    stats = tools.Statistics(lambda ind: ind.fitness.values[0])  # Initializes statistics tracking
    stats.register("avg", lambda x: round(np.mean(x), 5))  # Registers the average fitness statistic with rounding
    stats.register("min", lambda x: round(np.min(x), 5))  # Registers the minimum fitness statistic with rounding
    stats.register("max", lambda x: round(np.max(x), 5))  # Registers the maximum fitness statistic with rounding

    logbook = tools.Logbook()
    logbook.header = ['gen', 'nevals'] + (stats.fields if stats else [])

    # Evaluate the entire initial population
    invalid_ind = [ind for ind in pop if not ind.fitness.valid]
    fitnesses = map(toolbox.evaluate, invalid_ind)
    for ind, fit in zip(invalid_ind, fitnesses):
        ind.fitness.values = fit

    record = stats.compile(pop) if stats else {}
    logbook.record(gen=0, nevals=len(invalid_ind), **record)
    print(logbook.stream)

    # Initialize variables for stall generation limit
    best_fitness = -1
    no_improvement_count = 0

    for gen in range(1, n_generation + 1):
        # Apply elitism: retain the best elite_count individuals
        elite_individuals = toolbox.elitism(pop)

        # Select the top 50% individuals based on fitness from the remaining population
        remaining_population = [ind for ind in pop if ind not in elite_individuals]
        top_50_percent = toolbox.select(remaining_population, len(remaining_population) // 2)
        top_50_percent = list(map(toolbox.clone, top_50_percent))

        # Ensure population size is correct after elitism
        offspring = []
        while len(offspring) < n_population - len(elite_individuals) - len(top_50_percent):
            if len(top_50_percent) < 2:
                break
            parent1, parent2 = deap_random.sample(top_50_percent, 2)
            child1, child2 = toolbox.clone(parent1), toolbox.clone(parent2)
            if deap_random.random() < cxpb:
                toolbox.mate(child1, child2)
                del child1.fitness.values
                del child2.fitness.values
            offspring.append(child1)
            offspring.append(child2)

        # Apply mutation on the offspring
        for mutant in offspring:
            if deap_random.random() < mutpb:
                toolbox.mutate(mutant)
                del mutant.fitness.values

        # Evaluate the individuals with an invalid fitness
        invalid_ind = [ind for ind in offspring if not ind.fitness.valid]
        fitnesses = map(toolbox.evaluate, invalid_ind)
        for ind, fit in zip(invalid_ind, fitnesses):
            ind.fitness.values = fit

        # Combine elites, top 50% and their offspring to form the new population
        pop[:] = elite_individuals + top_50_percent + offspring

        # Ensure population size remains constant
        pop = pop[:n_population]

        # Update the hall of fame with the generated individuals
        hof.update(pop)

        # Update statistics
        record = stats.compile(pop)
        logbook.record(gen=gen, nevals=len(invalid_ind), **record)
        print(logbook.stream)

        # Check for stall generation limit
        current_best_fitness = max([ind.fitness.values[0] for ind in pop])
        if current_best_fitness > best_fitness:
            best_fitness = current_best_fitness
            no_improvement_count = 0
        else:
            no_improvement_count += 1

        if no_improvement_count >= stall_gen_limit:
            print(f"Stopping early at generation {gen} due to no improvement for {stall_gen_limit} consecutive generations.")
            print()
            break

    return hof, logbook

This function identifies the best subset of features from the Hall of Fame (HOF) based on the maximum fitness value and calculates its metrics using the base classifiers.

Parameters:

  • hof: Hall of Fame containing the best individuals found by the genetic algorithm.
  • X: The feature matrix.
  • y: The target vector.
  • clf: base classifiers
In[7]:
def bestIndividual(hof, X, y, clf):
    maxAccuracy = 0.0
    best_sensitivity = 0.0
    best_specificity = 0.0
    best_precision = 0.0
    best_recall = 0.0
    best_f1 = 0.0
    _individual = None

    for individual in hof:
        if individual.fitness.values[0] > maxAccuracy:
            maxAccuracy = individual.fitness.values[0]
            _individual = individual

    if _individual:
        _individualHeader = [i for i in range(len(_individual)) if _individual[i] == 1]
        X_selected = X[:, _individualHeader]
        clf.fit(X_selected, y)
        y_pred = cross_val_predict(clf, X_selected, y, cv=5)
        sensitivity, specificity, precision, recall, f1 = calculate_metrics(y, y_pred)
        return maxAccuracy, _individual, _individualHeader, sensitivity, specificity, precision, recall, f1

    return 0, None, [], 0, 0, 0, 0, 0

In the following chunk of code, first, we initialize the loop to iterates over each classifier defined in the 'classifiers' dictionary. Next, we run the geneticAlgorithm() function from above. After calling the GA function, we perform k-fold cross validation on the training data.

In each fold, the classifier is trained on the selected features of the training fold and it predicts on the validation fold to generate meta-features (stored in 'meta_features').

Next we calculate the accuracy, sensitivity, specificity, precision, recall, and f1 of the classifier on the test set. The results are appended to result list.

In[13]:
seed_value = 22
random.seed(seed_value)
np.random.seed(seed_value)

for idx, (clf_name, clf) in enumerate(classifiers.items()):
    print(f"Running GA for {clf_name}")
    
    # Get the best feature subset for the current classifier from GA
    hof, logbook = geneticAlgorithm(X_train_scaled, y_train, clf, n_population, n_generation, cxpb, mutpb, elite_count, stall_gen_limit, seed=seed_value)
    accuracy, individual, selected_features, sensitivity, specificity, precision, recall, f1 = bestIndividual(hof, X_train_scaled, y_train, clf)
    
    # Train the classifier on the selected features using cross-validation
    meta_features = np.zeros((X_train_scaled.shape[0],))
    for train_idx, valid_idx in kf.split(X_train_scaled):
        X_train_fold, X_valid_fold = X_train_scaled[train_idx], X_train_scaled[valid_idx]
        y_train_fold, y_valid_fold = y_train.to_numpy()[train_idx], y_train.to_numpy()[valid_idx]  # Convert to numpy arrays

        # Train the classifier on the selected features
        X_train_selected = X_train_fold[:, selected_features]
        clf.fit(X_train_selected, y_train_fold)

        # Generate meta-features
        X_valid_selected = X_valid_fold[:, selected_features]
        meta_features[valid_idx] = clf.predict(X_valid_selected)
    
    stacked_features[:, idx] = meta_features
    
    # Train on the entire training data for the test set
    X_train_selected = X_train_scaled[:, selected_features]
    X_test_selected = X_test_scaled[:, selected_features]
    clf.fit(X_train_selected, y_train)
    stacked_test_features[:, idx] = clf.predict(X_test_selected)
    
    # Store the trained classifier
    base_classifiers.append((clf_name, clf))

    # Store the results for this base classifier
    accuracy = accuracy_score(y_test, stacked_test_features[:, idx])
    sensitivity, specificity, precision, recall, f1 = calculate_metrics(y_test, stacked_test_features[:, idx])
    results.append({
        'Classifier Name': clf_name,
        'Individual': individual,
        'Feature Subset': selected_features,
        'Accuracy': accuracy,
        'Sensitivity': sensitivity,
        'Specificity': specificity,
        'Precision': precision,
        'Recall': recall,
        'F1 Score': f1
    })

def calculate_sensitivity_specificity_f1(y_true, y_pred):
    tn, fp, fn, tp = confusion_matrix(y_true, y_pred).ravel()
    sensitivity = tp / (tp + fn) if (tp + fn) > 0 else 0
    specificity = tn / (tn + fp) if (tn + fp) > 0 else 0
    f1 = f1_score(y_true, y_pred)
    return sensitivity, specificity, f1

def evaluate_classifier(clf, X_train, y_train, X_test, y_test):
    clf.fit(X_train, y_train)
    y_pred = clf.predict(X_test)
    accuracy = accuracy_score(y_test, y_pred)
    sensitivity, specificity, f1 = calculate_sensitivity_specificity_f1(y_test, y_pred)
    return accuracy, sensitivity, specificity, f1

def evaluate_kfold(clf, X, y, k):
    kf = KFold(n_splits=k)
    accuracies = []
    sensitivities = []
    specificities = []
    f1_scores = []
    for train_idx, test_idx in kf.split(X):
        X_train, X_test = X[train_idx], X[test_idx]
        y_train, y_test = y.to_numpy()[train_idx], y.to_numpy()[test_idx]  # Convert y to numpy arrays
        clf.fit(X_train, y_train)
        y_pred = clf.predict(X_test)
        accuracy = accuracy_score(y_test, y_pred)
        sensitivity, specificity, f1 = calculate_sensitivity_specificity_f1(y_test, y_pred)
        accuracies.append(accuracy)
        sensitivities.append(sensitivity)
        specificities.append(specificity)
        f1_scores.append(f1)
    return np.mean(accuracies) * 100, np.mean(sensitivities) * 100, np.mean(specificities) * 100, np.mean(f1_scores) * 100
